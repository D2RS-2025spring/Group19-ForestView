import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA
from sklearn.metrics import mean_squared_error
from sklearn.model_selection import train_test_split
import warnings
import os
warnings.filterwarnings('ignore')

# 
plt.rcParams["font.family"] = ["SimHei", "WenQuanYi Micro Hei"]  # 
plt.rcParams['axes.unicode_minus'] = False  # 

# 
output_dir = r"C:\Users\PC\Desktop\"
os.makedirs(output_dir, exist_ok=True)

# 
df = pd.read_csv(r"C:\Users\PC\Desktop\\2013-2023monthly_avg_humidity.csv")

# 
df['date'] = pd.to_datetime(df['date'])

# 
df['humidity'] = pd.to_numeric(df['humidity'])

# 
df.set_index('date', inplace=True)

# 
print(":")
df.info()

#

df['humidity'].fillna(method='ffill', inplace=True)

# ======================  2 ======================
from statsmodels.tsa.seasonal import seasonal_decompose
decomposition = seasonal_decompose(df['humidity'], model='additive', period=12)
trend, seasonal, residual = decomposition.trend, decomposition.seasonal, decomposition.resid

plt.figure(figsize=(12, 10))
plt.suptitle('', y=0.95, fontsize=14)  # 

plt.subplot(411)

plt.legend()

# ======================  ======================
train_size = int(len(df) * 0.8)
train, test = df.iloc[:train_size], df.iloc[train_size:]

# ====================== ARIMA ======================
model = ARIMA(train['humidity'], order=(1, 1, 1), seasonal_order=(1, 1, 1, 12))
model_fit = model.fit()

# 
predictions = model_fit.forecast(steps=len(test))
predictions.index = test.index

# RMSE
rmse = np.sqrt(mean_squared_error(test['humidity'], predictions))
print(f'\nRMSE: {rmse:.2f}%')

# ======================  3 ======================
plt.figure(figsize=(12, 6))
plt.plot(train['humidity'], label='', alpha=0.6)
plt.plot(test['humidity'], label='', alpha=0.6)
