import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA
from sklearn.metrics import mean_squared_error
from sklearn.model_selection import train_test_split
import warnings
import os
warnings.filterwarnings('ignore')

# 
plt.rcParams["font.family"] = ["SimHei", "WenQuanYi Micro Hei", "Heiti TC"]
plt.rcParams['axes.unicode_minus'] = False  # 

# 
output_dir = r"C:\Users\PC\Desktop\"
os.makedirs(output_dir, exist_ok=True)

# 
df = pd.read_csv(r"C:\Users\PC\Desktop\\2013-2023monthly_avg_sealevelpressure.csv")

# 
df['date'] = pd.to_datetime(df['date'])

# 
df['sealevelpressure'] = pd.to_numeric(df['sealevelpressure'])

# 
df.set_index('date', inplace=True)

# 
print(":")
df.info()

# 
print("\n:")
print(df.isnull().sum())

# 
df['sealevelpressure'].fillna(method='ffill', inplace=True)

#  - 
plt.figure(figsize=(12, 6))
df['sealevelpressure'].plot(title='2013-2023')
plt.xlabel('')
plt.ylabel(' (hPa)')
plt.grid(True)
plt.tight_layout()
plt.savefig(os.path.join(output_dir, '1_.png'))
plt.show()

# 
from statsmodels.tsa.seasonal import seasonal_decompose
decomposition = seasonal_decompose(df['sealevelpressure'], model='additive', period=12)
trend = decomposition.trend
seasonal = decomposition.seasonal
residual = decomposition.resid

#  - 
plt.figure(figsize=(12, 10))
plt.subplot(411)
plt.plot(df['sealevelpressure'], label='')
plt.title('')
plt.legend(loc='best')
plt.subplot(412)
plt.plot(trend, label='')
plt.legend(loc='best')
plt.subplot(413)
plt.plot(seasonal, label='')
plt.legend(loc='best')
plt.subplot(414)
plt.plot(residual, label='')
plt.legend(loc='best')
plt.tight_layout()
plt.savefig(os.path.join(output_dir, '2_.png'))
plt.show()

# 
train_size = int(len(df) * 0.8)
train, test = df.iloc[:train_size], df.iloc[train_size:]

# ARIMA
# AIC/BIC
model = ARIMA(train['sealevelpressure'], order=(1, 1, 1), seasonal_order=(1, 1, 1, 12))
model_fit = model.fit()

# 
predictions = model_fit.forecast(steps=len(test))
predictions.index = test.index

# RMSE
rmse = np.sqrt(mean_squared_error(test['sealevelpressure'], predictions))
print(f'\nRMSE: {rmse:.2f} hPa')

#  - 
plt.figure(figsize=(12, 6))
plt.plot(train['sealevelpressure'], label='')
plt.plot(test['sealevelpressure'], label='')
plt.plot(predictions, label='', color='r')
plt.title('')
plt.xlabel('')
plt.ylabel(' (hPa)')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.savefig(os.path.join(output_dir, '3_.png'))
plt.show()

# 12
future_predictions = model_fit.forecast(steps=12)
future_dates = pd.date_range(start=df.index[-1], periods=13, freq='MS')[1:]
future_predictions.index = future_dates

#  - 
plt.figure(figsize=(12, 6))
plt.plot(df['sealevelpressure'], label='')
plt.plot(future_predictions, label='12', color='g', linestyle='--')
plt.title('12')
plt.xlabel('')
plt.ylabel(' (hPa)')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.savefig(os.path.join(output_dir, '4_.png'))
plt.show()

# 
save_path = os.path.join(output_dir, '12.csv')
future_df = pd.DataFrame({
    'date': future_dates,
    'predicted_sealevelpressure': future_predictions
})
future_df.to_csv(save_path, index=False, encoding='utf-8-sig')
print(f"\n12 '{save_path}'")
